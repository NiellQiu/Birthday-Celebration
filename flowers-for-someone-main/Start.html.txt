    <script>
        function startMusic() {
            // Create and play audio
            let audio = new Audio("like_a_dream.mp3");
            audio.loop = true;
            audio.play();

            // Save music instance
            sessionStorage.setItem("musicPlaying", "true");
            sessionStorage.setItem("musicTime", "0");

            // Redirect to index.html
            window.location.href = "index.html";
        }
    </script>
